### 2. ThreeKingdomsEngine.ts — giveRes() 无上限检查

```typescript
// 当前：资源永不截断
private giveRes(id: string, amt: number): void {
  this.res[id] = Math.max(0, (this.res[id] || 0) + amt);
}

// 应改为：检查资源上限
private giveRes(id: string, amt: number): void {
  const current = this.res[id] || 0;
  const cap = this.getResourceCap(id);
  if (cap > 0 && current >= cap && amt > 0) return; // 已满，停止
  this.res[id] = Math.max(0, Math.min(cap > 0 ? cap : Infinity, current + amt));
}
```

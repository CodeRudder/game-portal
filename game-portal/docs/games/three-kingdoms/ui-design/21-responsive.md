## 二、画布缩放详细算法

### 2.1 核心缩放公式

```javascript
/**
 * 画布缩放计算
 * @param {number} viewW - 视口宽度
 * @param {number} viewH - 视口高度
 * @returns {{ scale, offsetX, offsetY, canvasW, canvasH }}
 */
function calcCanvasScale(viewW, viewH) {
  const BASE_W = 1280;
  const BASE_H = 800;
  const isMobile = viewW < 768;

  if (isMobile) {
    // 移动端：不使用画布缩放，采用流式布局
    return { scale: 1, offsetX: 0, offsetY: 0, canvasW: viewW, canvasH: viewH };
  }

  // PC/平板：等比缩放
  const scaleW = viewW / BASE_W;
  const scaleH = viewH / BASE_H;
  const scale = Math.min(scaleW, scaleH);

  // 限制最大缩放（4K下不超过2倍，避免元素过大）
  const clampedScale = Math.min(scale, 2.0);

  const canvasW = Math.round(BASE_W * clampedScale);
  const canvasH = Math.round(BASE_H * clampedScale);

  // 居中偏移
  const offsetX = Math.round((viewW - canvasW) / 2);
  const offsetY = Math.round((viewH - canvasH) / 2);

  return { scale: clampedScale, offsetX, offsetY, canvasW, canvasH };
}
```

### 2.2 留白区域处理

| 缩放情况 | 留白位置 | 处理策略 |
|---------|---------|---------|
| 宽屏（16:9 显示器） | 上下留白 | 填充深色水墨纹理背景 `#0a0a14`，上方叠加远山剪影 SVG |
| 窄屏（4:3 平板） | 两侧留白 | 填充同色水墨纹理，叠加竹叶/云纹装饰 |
| 超宽屏（21:9） | 两侧大面积留白 | 展开辅助面板（战报/武将信息），不浪费空间 |
| 4K（≥3840） | 整体放大 | scale 限幅 2.0，剩余空间展示装饰背景 |

### 2.3 缩放应用方式

```css
/* PC/平板端：CSS Transform 缩放 */
#game-canvas-wrapper {
  position: absolute;
  width: 1280px;
  height: 800px;
  transform-origin: top left;
  /* JS 动态设置 transform: scale(X) translate(Xpx, Ypx) */
}

/* 移动端：流式布局，不使用 transform */
@media (max-width: 767px) {
  #game-canvas-wrapper {
    width: 100%;
    height: auto;
    transform: none;
  }
}
```

function detectPerformanceTier() {
  const cores = navigator.hardwareConcurrency || 2;
  const memory = navigator.deviceMemory || 2; // Chrome only

## 八、横竖屏切换方案（手机端）

### 8.1 切换检测

```javascript
// 双重检测：resize + orientation media query
let currentOrientation = 'portrait';

function detectOrientation() {
  const isLandscape = window.innerWidth > window.innerHeight;
  return isLandscape ? 'landscape' : 'portrait';
}

// 监听方式1：resize（最可靠）
window.addEventListener('resize', debounce(() => {
  const newOrientation = detectOrientation();
  if (newOrientation !== currentOrientation) {
    onOrientationChange(currentOrientation, newOrientation);
    currentOrientation = newOrientation;
  }
}, 150));

// 监听方式2：screen.orientation（补充）
if (screen.orientation) {
  screen.orientation.addEventListener('change', () => {
    const newOrientation = detectOrientation();
    if (newOrientation !== currentOrientation) {
      onOrientationChange(currentOrientation, newOrientation);
      currentOrientation = newOrientation;
    }
  });
}
```
